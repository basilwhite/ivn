def sanitize_df_for_excel(df: pd.DataFrame, max_cell_length: int = 32000) -> pd.DataFrame:
    """Return a copy of `df` with values converted to Excel-safe types/strings.

    - Replaces NaN/NaT with empty string
    - Converts lists/dicts/tuples to JSON strings
    - Converts numpy scalars to Python scalars
    - Removes XML-invalid control characters
    - Truncates very long strings to `max_cell_length` (Excel limit ~32,767)
    """
    if df is None:
        return df

    control_chars_re = re.compile(r"[\x00-\x08\x0B\x0C\x0E-\x1F]")

    def sanitize_value(v):
        # pandas NA / numpy nan / None
        try:
            if pd.isna(v):
                return ""
        except Exception:
            pass

        # JSON-serialize common container types
        if isinstance(v, (list, dict, tuple)):
            try:
                s = _json.dumps(v, ensure_ascii=False)
            except Exception:
                s = str(v)
            s = control_chars_re.sub('', s)
            return s[:max_cell_length]

        # numpy scalars
        try:
            import numpy as _np
            if isinstance(v, _np.generic):
                return v.item()
        except Exception:
            pass

        # datetimes -> ISO
        try:
            if isinstance(v, (datetime.date, datetime.datetime)):
                return v.isoformat()
        except Exception:
            pass

        # Everything else: stringify and clean control chars
        s = str(v)
        s = control_chars_re.sub('', s)
        if len(s) > max_cell_length:
            return s[:max_cell_length]
        return s

    # Apply sanitize_value elementwise
    try:
        safe_df = df.copy()
        for col in safe_df.columns:
            # Apply per-column to avoid dtype coercion surprises
            safe_df[col] = safe_df[col].apply(sanitize_value)
        return safe_df
    except Exception:
        # In case of unexpected types, fall back to coarse conversion
        try:
            return df.astype(str).replace('nan', '')
        except Exception:
            return df

def ask_file_path_from_list(script_dir: str):
    json_files = sorted([f for f in os.listdir(script_dir) if f.startswith("crosswalk_inferences_training(") and f.endswith(".json")])
    if not json_files:
        print_verbose("No training JSON files found in the script folder.")
        return None
    print("Select a training JSON file:")
    for idx, fname in enumerate(json_files, 1):
        print(f"{idx}: {fname}")
    print("0: Run without a training file")
    while True:
        choice = input("Enter the number of the file to use (or 0): ").strip()
        if choice == "0":
            return None
        try:
            idx = int(choice)
            if 1 <= idx <= len(json_files):
                return str(Path(script_dir) / json_files[idx - 1])
        except Exception:
            pass
        print("Invalid selection. Please try again.")


def similar(a: str, b: str) -> float:
    return SequenceMatcher(None, str(a), str(b)).ratio()

def build_alignment_graph(alignments_df):
    G = nx.Graph()
    if alignments_df is not None:
        for idx, row in alignments_df.iterrows():
            # Use correct columns from Alignments sheet12
            a = str(row.get("Enabling Component", row.get("component_name", ""))).strip()
            b = str(row.get("Dependent Component", row.get("Component", ""))).strip()
            if a and b:
                G.add_edge(a, b)
    return G


def extract_features(pairs, graph):
    features = []
    for a, b in pairs:
        str_sim = similar(a, b)
        try:
            path_length = nx.shortest_path_length(graph, a, b)
            indirect_strength = 1.0 / (path_length + 1)
        except (nx.NodeNotFound, nx.NetworkXNoPath):
            indirect_strength = 0.0
        features.append([str_sim, indirect_strength])
    return np.array(features)


def verify_component_source_alignment(results_df, components_df):
    """Verify that components in results maintain their original source alignment from Components tab.
    
    Returns a dict with verification results:
    {
        "total_rows": int,
        "verified_enabling": int,
        "unverified_enabling": int,
        "mismatched_enabling": int,
        "verified_dependent": int,
        "unverified_dependent": int,
        "issues": [list of issue descriptions]
    }
    """
    if results_df is None or components_df is None or len(results_df) == 0:
        return {"total_rows": 0, "verified_enabling": 0, "unverified_enabling": 0, 
                "mismatched_enabling": 0, "verified_dependent": 0, "unverified_dependent": 0, "issues": []}
    
    # Build lookup of component -> source from Components tab
    comp_source_lookup = {}
    for idx, row in components_df.iterrows():
        comp_name = str(row.get("Component", row.get("component_name", ""))).strip()
        source_id = str(row.get("Source", row.get("source_id", ""))).strip()
        if comp_name and source_id:
            comp_source_lookup[comp_name] = source_id
    
    verified_enabling = 0
    unverified_enabling = 0
    mismatched_enabling = 0
    verified_dependent = 0
    unverified_dependent = 0
    issues = []
    
    for idx, row in results_df.iterrows():
        enabling_comp = str(row.get("Enabling Component", "")).strip()
        enabling_source = str(row.get("Enabling Source", "")).strip()
        dependent_comp = str(row.get("Dependent Component", "")).strip()
        dependent_source = str(row.get("Dependent Source", "")).strip()
        
        # Verify enabling component
        if enabling_comp in comp_source_lookup:
            expected_source = comp_source_lookup[enabling_comp]
            if enabling_source == expected_source:
                verified_enabling += 1
            else:
                mismatched_enabling += 1
                issues.append(f"Row {idx+2}: Enabling Component '{enabling_comp}' has source '{enabling_source}' but Components tab shows '{expected_source}'")
        else:
            unverified_enabling += 1
        
        # Verify dependent component if it exists in Components tab; otherwise check presence
        if dependent_comp in comp_source_lookup:
            expected_dep_source = comp_source_lookup[dependent_comp]
            if dependent_source == expected_dep_source:
                verified_dependent += 1
            else:
                mismatched_dependent = mismatched_dependent + 1 if 'mismatched_dependent' in locals() else 1
                issues.append(f"Row {idx+2}: Dependent Component '{dependent_comp}' has source '{dependent_source}' but Components tab shows '{expected_dep_source}'")
        else:
            if dependent_source:
                verified_dependent += 1
            else:
                unverified_dependent += 1
                if dependent_comp:
                    issues.append(f"Row {idx+2}: Dependent Component '{dependent_comp}' has no Dependent Source")
    
    # Ensure mismatched_dependent is present
    mismatched_dependent = mismatched_dependent if 'mismatched_dependent' in locals() else 0
    return {
        "total_rows": len(results_df),
        "verified_enabling": verified_enabling,
        "unverified_enabling": unverified_enabling,
        "mismatched_enabling": mismatched_enabling,
        "verified_dependent": verified_dependent,
        "unverified_dependent": unverified_dependent,
        "mismatched_dependent": mismatched_dependent,
        "issues": issues
    }


def enforce_component_metadata(results_df, components_df, tobecrosswalked_df, sources_df):
    """Ensure each component in results_df retains original metadata from Components (preferred) or ToBeCrosswalked.

    Overwrites fields in results_df for: Source, Component Description, Component URL, Office of Primary Interest, Source Agency.
    Returns (new_df, report_dict)
    """
    if results_df is None or len(results_df) == 0:
        return results_df, {"total": 0, "enforced": 0, "issues": []}

    # Build component -> metadata from Components (preferred)
    comp_meta = {}
    if components_df is not None:
        for _, row in components_df.iterrows():
            name = str(row.get("Component", row.get("component_name", ""))).strip()
            if not name:
                continue
            source_id = str(row.get("Source", row.get("source_id", ""))).strip()
            description = str(row.get("Component Description", row.get("component_description", ""))).strip()
            url = str(row.get("Component URL", row.get("component_url", ""))).strip()
            office = str(row.get("Component Office of Primary Interest", row.get("component_ofc_of_primary_interest", row.get("component_ofc_of_primary_interest", "")))).strip()
            comp_meta[name] = {
                "source_id": source_id,
                "description": description,
                "url": url,
                "office": office,
                "origin": "Components"
            }

    # Build fallback metadata from ToBeCrosswalked
    if tobecrosswalked_df is not None:
        for _, row in tobecrosswalked_df.iterrows():
            name = str(row.get("Component", "")).strip()
            if not name or name in comp_meta:
                continue
            source_id = str(row.get("Source", "")).strip()
            description = str(row.get("Component Description", "")).strip()
            url = str(row.get("Component URL", "")).strip()
            office = str(row.get("Component Office of Primary Interest", "")).strip()
            comp_meta[name] = {
                "source_id": source_id,
                "description": description,
                "url": url,
                "office": office,
                "origin": "ToBeCrosswalked"
            }

    # Build source agency lookup from sources_df
    source_agency_lookup = {}
    if sources_df is not None:
        for _, row in sources_df.iterrows():
            sid = str(row.get("source_id", row.get("Source", ""))).strip()
            if not sid:
                continue
            # try several plausible keys for agency
            agency = row.get("source_agency") if "source_agency" in row else row.get("Source Agency") if "Source Agency" in row else row.get("agency") if "agency" in row else row.get("agency_name") if "agency_name" in row else ""
            source_agency_lookup[sid] = agency

    df = results_df.copy()
    enforced = 0
    issues = []
    for idx, row in df.iterrows():
        # Enabling
        en_comp = str(row.get("Enabling Component", "")).strip()
        if en_comp and en_comp in comp_meta:
            meta = comp_meta[en_comp]
            # Overwrite fields
            if str(row.get("Enabling Source", "")).strip() != meta.get("source_id", ""):
                df.at[idx, "Enabling Source"] = meta.get("source_id", "")
                enforced += 1
            df.at[idx, "Enabling Component Description"] = meta.get("description", "")
            df.at[idx, "Enabling Component URL"] = meta.get("url", "")
            df.at[idx, "Enabling Component Office of Primary Interest"] = meta.get("office", "")
            # Source agency
            sid = meta.get("source_id", "")
            df.at[idx, "Enabling Source Agency"] = source_agency_lookup.get(sid, "")

        # Dependent
        de_comp = str(row.get("Dependent Component", "")).strip()
        if de_comp and de_comp in comp_meta:
            meta = comp_meta[de_comp]
            if str(row.get("Dependent Source", "")).strip() != meta.get("source_id", ""):
                df.at[idx, "Dependent Source"] = meta.get("source_id", "")
                enforced += 1
            df.at[idx, "Dependent Component Description"] = meta.get("description", "")
            df.at[idx, "Dependent Component URL"] = meta.get("url", "")
            df.at[idx, "Dependent Component Office of Primary Interest"] = meta.get("office", "")
            sid = meta.get("source_id", "")
            df.at[idx, "Dependent Source Agency"] = source_agency_lookup.get(sid, "")

    report = {"total": len(df), "enforced": enforced, "issues": issues}
    return df, report


def build_pairs_to_predict(components_df, tobecrosswalked_df, sources_df=None):
    """Build the full list of candidate pairs and metadata (same logic as infer_alignments).
    Returns (pairs_list, pair_metadata_list).
    """
    pairs_to_predict = []
    pair_metadata = []
    if components_df is None or tobecrosswalked_df is None:
        return pairs_to_predict, pair_metadata

    comp_rows = components_df.to_dict("records")
    tobe_rows = tobecrosswalked_df.to_dict("records")

    for comp in comp_rows:
        for tobe in tobe_rows:
            comp_source = str(comp.get("Source", comp.get("source_id", ""))).strip()
            tobe_source = str(tobe.get("Source", "")).strip()
            # Skip if both components are from the same source
            if comp_source and tobe_source and comp_source == tobe_source:
                continue
            enabling = str(comp.get("Component", comp.get("component_name", ""))).strip()
            dependent = str(tobe.get("Component", "")).strip()
            if not enabling or not dependent:
                continue
            pairs_to_predict.append((enabling, dependent))
            pair_metadata.append({
                "Enabling Component": enabling,
                "Dependent Component": dependent
            })
    return pairs_to_predict, pair_metadata


def estimate_threshold_from_sample(pairs, model, sample_size=10000, target_rows=90000):
    """Estimate a confidence threshold using a sample of candidate pairs.

    Uses deterministic classification (if available) to estimate thresholds.
    Applies a conservative 1.5x safety factor to ensure estimates don't exceed actual results.

    Returns (estimated_threshold, estimated_rows_at_threshold, sample_size_used, percentile_estimates)
    """
    total_pairs = len(pairs)
    if total_pairs == 0:
        return 0.0, 0, 0, {}

    sample_n = min(sample_size, total_pairs)
    # Use a random sample without replacement to avoid ordering bias
    rng = np.random.default_rng(seed=42)
    indices = rng.choice(total_pairs, size=sample_n, replace=False)
    sample_pairs = [pairs[i] for i in indices]

    # Build graph from model (used by feature extraction)
    graph = nx.Graph()
    graph.add_edges_from(model.get("graph_edges", []))

    # Extract features for the sample (not scaled using a proper scaler here,
    # because the inference uses a simplified random prediction; we keep the
    # feature extraction to maintain parity if you later improve prediction.)
    X_sample = extract_features(sample_pairs, graph)

    # If a saved classifier is available in the model, use it for deterministic probabilities
    clf_obj = model.get("clf_obj")
    if clf_obj is not None and "clf" in clf_obj and "scaler" in clf_obj:
        scaler = clf_obj["scaler"]
        clf = clf_obj["clf"]
        Xs = extract_features(sample_pairs, graph)
        try:
            Xs_scaled = scaler.transform(Xs)
            # Predict probabilities for the positive class
            sample_conf = clf.predict_proba(Xs_scaled)[:, 1]
        except Exception:
            # Fallback to random if scaler/clf fail for any reason
            sample_conf = np.random.rand(len(sample_pairs))
    else:
        # Generate sample predictions using the same simplified method as infer_alignments
        sample_conf = np.random.rand(len(sample_pairs))

    # Estimate threshold such that estimated rows <= target_rows
    ratio = float(target_rows) / float(total_pairs)
    if ratio >= 1.0:
        est_threshold = 0.0
    else:
        quantile = max(0.0, 1.0 - ratio)
        est_threshold = float(np.quantile(sample_conf, quantile))

    est_frac = float((sample_conf >= est_threshold).sum()) / float(len(sample_conf))
    est_rows = int(round(est_frac * total_pairs))

    # Apply a conservative 1.5x safety factor to the estimated rows to account for sampling variance
    # This ensures we undershoot the actual result and never exceed Excel limits
    SAFETY_FACTOR = 1.5
    est_rows_safe = int(round(est_rows * SAFETY_FACTOR))

    # Compute some percentile-based candidate thresholds and their estimated row counts
    percentiles = [0.5, 0.75, 0.9, 0.95, 0.99]
    pct_estimates = {}
    for p in percentiles:
        thr = float(np.quantile(sample_conf, p))
        frac = float((sample_conf >= thr).sum()) / float(len(sample_conf))
        rows = int(round(frac * total_pairs))
        rows_safe = int(round(rows * SAFETY_FACTOR))
        pct_estimates[p] = {"threshold": thr, "estimated_rows": rows, "estimated_rows_safe": rows_safe}

    return est_threshold, est_rows_safe, sample_n, pct_estimates


def train_alignment_model(alignments_df, nonaligned_df, components_list):
    start_time = time.time()
    model = {"threshold": 0.6}
   
    # Build alignment graph
    print_verbose("Building alignment graph...")
    graph = build_alignment_graph(alignments_df)
    model["graph_edges"] = list(graph.edges())
    print_verbose(f"Graph built with {len(graph.edges())} edges - Elapsed: {format_time(time.time() - start_time)}")
   
    # Prepare training data
    print_verbose("Preparing training data...")
    pairs, labels = [], []
    # Use Dataset for aligned pairs
    if alignments_df is not None:
        total_aligned = len(alignments_df)
        for idx, row in alignments_df.iterrows():
            a = str(row.get("Enabling Component", "")).strip()
            b = str(row.get("Dependent Component", "")).strip()
            if a and b:
                pairs.append((a, b))
                labels.append(1)
        print_verbose(f"Processed {len([l for l in labels if l == 1])} aligned pairs from {total_aligned} rows")
    
    # Use Nonaligned-Edge-Cases for nonaligned pairs
    if nonaligned_df is not None:
        total_nonaligned = len(nonaligned_df)
        for idx, row in nonaligned_df.iterrows():
            a = str(row.get("Enabling Component", "")).strip()
            b = str(row.get("Dependent Component", "")).strip()
            if a and b:
                pairs.append((a, b))
                labels.append(0)
        print_verbose(f"Processed {len([l for l in labels if l == 0])} nonaligned pairs from {total_nonaligned} rows")
   
    print_verbose(f"Total training pairs: {len(pairs)} - Elapsed: {format_time(time.time() - start_time)}")
   
    if not pairs:
        print_verbose("No training pairs found.")
        return model
   
    # Extract features
    print_verbose(f"Extracting features for {len(pairs)} training pairs...")
    feature_start = time.time()
    X = extract_features(pairs, graph)
    y = np.array(labels)
    print_verbose(f"Feature extraction complete - Elapsed: {format_time(time.time() - feature_start)}")
     # Train classifier
    print_verbose("Training classifier...")
    train_start = time.time()
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    clf = RandomForestClassifier(n_estimators=100, random_state=42)
    clf.fit(X_scaled, y)
    print_verbose(f"Classifier training complete - Elapsed: {format_time(time.time() - train_start)}")
    
    model["classifier"] = {
        "scaler_mean": scaler.mean_.tolist(),
        "scaler_scale": scaler.scale_.tolist(),
        "classes": clf.classes_.tolist() if hasattr(clf.classes_, "tolist") else list(clf.classes_),
        "feature_importances": clf.feature_importances_.tolist() if hasattr(clf.feature_importances_, "tolist") else list(clf.feature_importances_),
        "n_features_in": int(clf.n_features_in_),
        "n_classes": int(clf.n_classes_),
        "estimators": [est.tree_.__getstate__() for est in clf.estimators_]
    }

    # Also attach full classifier and scaler for deterministic predictions (and saveable via joblib)
    try:
        model["clf_obj"] = {"clf": clf, "scaler": scaler}
    except Exception:
        model["clf_obj"] = None

    print_verbose(f"Model training complete - Total time: {format_time(time.time() - start_time)}")

    return model


def convert_ndarrays(obj):
    if isinstance(obj, np.ndarray):
        return obj.tolist()
    if isinstance(obj, dict):
        return {k: convert_ndarrays(v) for k, v in obj.items()}
    if isinstance(obj, list):
        return [convert_ndarrays(v) for v in obj]
    return obj


def save_model_json(model, out_path):

            for name, df in all_tabs.items():
                try:
                    base = name.replace(" ", "_")[:31]
                    sheet_name = base
                    suffix = 1
                    # Ensure uniqueness within this workbook (Excel requires unique sheet names)
                    while sheet_name in used_sheet_names:
                        # reserve space for suffix and keep <=31 chars
                        suffix_str = f"_{suffix}"
                        avail = 31 - len(suffix_str)
                        sheet_name = (base[:avail] + suffix_str)[:31]
                        suffix += 1
                    used_sheet_names.add(sheet_name)

                    try:
                        safe_df = sanitize_df_for_excel(df)
                    except Exception:
                        safe_df = df

                    safe_df.to_excel(writer, sheet_name=sheet_name, index=False)
                except Exception:
                    print_verbose(f"ERROR writing sheet '{name}' to output workbook. Skipping this sheet.")
                    traceback.print_exc()

            # Always write a verification issues sheet (header-only if there are no issues)
            try:
                sheet_name = 'verification_issues'[:31]
                if 'verification_issues_df' in locals() and isinstance(verification_issues_df, pd.DataFrame):
                    df_to_write = verification_issues_df
                    if df_to_write.empty:
                        # Ensure headers are present even if there are no rows
                        columns = ["row", "issue_type", "component", "field", "expected", "actual", "message"]
                        df_to_write = pd.DataFrame(columns=columns)
                        print_verbose("Verification issues DataFrame exists but is empty; writing header-only sheet.")
                    else:
                        print_verbose(f"Wrote verification issues sheet with {len(verification_issues_df)} records")
                else:
                    # No DataFrame was prepared; write an empty header-only sheet so the workbook always contains the tab
                    columns = ["row", "issue_type", "component", "field", "expected", "actual", "message"]
                    df_to_write = pd.DataFrame(columns=columns)
                    print_verbose("No verification issues DataFrame found; writing header-only verification issues sheet.")

                # Write the DataFrame (either populated or header-only)
                df_to_write.to_excel(writer, sheet_name=sheet_name, index=False)
            except Exception:
                print_verbose("ERROR writing verification_issues sheet to output workbook. Skipping this sheet.")
                traceback.print_exc()
        print_verbose("Processing complete.")
        print(f"Output file created at: {output_excel_path}")
    except Exception:
        print_verbose("ERROR while saving the Excel output file. See traceback below.")
        traceback.print_exc()
        raise
